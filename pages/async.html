Асинхронные функции в JavaScript

Асинхронные функции в JavaScript позволяют выполнять асинхронный код так, как будто он синхронный. Это достигается с помощью ключевых слов async и await.

Ключевое слово async

Ключевое слово async используется для объявления асинхронной функции, которая возвращает промис.

        async function fetchData() {
            return "Data fetched";
        }

        fetchData().then(data => console.log(data)); // Data fetched

Ключевое слово await

await используется для ожидания результата промиса. Оно может быть использовано только внутри асинхронной функции.

        async function fetchData() {
            let promise = new Promise((resolve, reject) => {
                setTimeout(() => resolve("Done!"), 1000)
            });

            let result = await promise; // ждет, пока промис не выполнится
            console.log(result); // "Done!"
        }

        fetchData();

Обработка ошибок

Для обработки ошибок в асинхронных функциях используется try...catch.

        async function fetchDataWithErrorHandling() {
            try {
                let response = await fetch('http://invalid-url');
                let data = await response.json();
                console.log(data);
            } catch (error) {
                console.log('Ошибка:', error);
            }
        }

        fetchDataWithErrorHandling();

Преимущества асинхронных функций

Асинхронные функции упрощают работу с асинхронным кодом, делая его более читаемым и понятным, как синхронный код.
